function prepReport(){
  var html=buildReportHTML();
  if(_rurl) URL.revokeObjectURL(_rurl);
  var blob=new Blob([html],{type:'text/html;charset=utf-8'});
  _rurl=URL.createObjectURL(blob);
  var lnk=document.getElementById('rlink');
  lnk.href=_rurl; lnk.className='export-link show';
  document.getElementById('rhint').className='export-info show';
}

function buildCR(){
  var t='AUDIOMÉTRIE TONALE ET VOCALE\n'+'─'.repeat(36)+'\n';
  if(g('nom')||g('prenom')) t+='Patient    : '+g('prenom')+' '+g('nom')+'\n';
  if(g('ddn')) t+='Né(e) le   : '+g('ddn')+'\n';
  if(g('dex')) t+='Examen     : '+g('dex')+'\n';
  if(g('examiner')) t+='Examinateur: '+g('examiner')+'\n';
  if(g('medecin')) t+='Prescr.    : '+g('medecin')+'\n';
  if(g('indic')) t+='Indication : '+g('indic')+'\n';
  if(g('atcd')) t+='Antécédents: '+g('atcd')+'\n';
  t+='\n';
  ['right','left'].forEach(function(ear){
    t+=(ear==='right'?'── OREILLE DROITE ──':'── OREILLE GAUCHE ──')+'\n';
    var ar=FREQS.map(function(f){return aD[ear].air[f]!==undefined?FLAB[FREQS.indexOf(f)]+':'+aD[ear].air[f]+'dB':null;}).filter(Boolean).join('  ');
    var ob=FREQS.map(function(f){return aD[ear].bone[f]!==undefined?FLAB[FREQS.indexOf(f)]+':'+aD[ear].bone[f]+'dB':null;}).filter(Boolean).join('  ');
    var mk=FREQS.map(function(f){return aD[ear].airMask[f]!==undefined?FLAB[FREQS.indexOf(f)]+':'+aD[ear].airMask[f]+'dB':null;}).filter(Boolean).join('  ');
    if(ar) t+='  Aérien  : '+ar+'\n';
    if(mk) t+='  Masquage: '+mk+'\n';
    if(ob) t+='  Osseux  : '+ob+'\n';
    var p=pta(ear); t+='  PTA     : '+(p!==null?p+' dB — '+grade(p):'—')+'\n';
    var suf=ear==='right'?'r':'l';
    var srt=g('srt-'+suf); if(srt) t+='  SRT     : '+srt+' dB HL'+(g('srtm-'+suf)?' (masq. '+g('srtm-'+suf)+' dB)':'')+'\n';
    var vl=VDBS.map(function(d){return vD[ear][d]!==undefined?d+'dB→'+vD[ear][d]+'%':null;}).filter(Boolean).join('  ');
    if(vl) t+='  Discr.  : '+vl+'\n';
    var td=tD[ear];
    if(td.type) t+='  Tymp.   : Type '+td.type+(g('tpres-'+suf)?' | P:'+g('tpres-'+suf)+' daPa':'')+'\n';
    t+='\n';
  });
  if(g('obs')) t+='Observations :\n'+g('obs')+'\n';
  return t;
}
function refreshCR(){ document.getElementById('crprev').textContent=buildCR(); }
function copyTxt(){
  var t=buildCR();
  if(navigator.clipboard){ navigator.clipboard.writeText(t).then(function(){alert('Copié !');}).catch(function(){fb(t);}); }
  else fb(t);
}
function fb(t){ var x=document.createElement('textarea');x.value=t;document.body.appendChild(x);x.select();try{document.execCommand('copy');}catch(e){}document.body.removeChild(x);alert('Copié !'); }

function resetAll(){
  if(!confirm('Effacer toutes les données ?')) return;
  ['right','left'].forEach(function(e){ aD[e]={air:{},bone:{},airMask:{}}; vD[e]={}; rD[e]={ipsi:{},contra:{}}; tD[e]={type:'',pres:'',comp:'',vol:''}; });
  document.querySelectorAll('.fi,.vi,.ri').forEach(function(i){i.value='';});
  document.querySelectorAll('.tymp-chip').forEach(function(c){c.classList.remove('sel');});
  ['nom','prenom','ddn','indic','atcd','acouph','obs','srt-r','srt-l','srtm-r','srtm-l','medecin','examiner','tpres-r','tpres-l','tcomp-r','tcomp-l','tvol-r','tvol-l'].forEach(function(id){var el=document.getElementById(id);if(el)el.value='';});
  document.getElementById('dex').valueAsDate=new Date();
  drawTonal(); drawVocal(); updatePTA(); updatePBMax();
  document.getElementById('rlink').className='export-link';
  document.getElementById('rhint').className='export-info';
}
